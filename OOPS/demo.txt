// ex :3 parameterized constructor

#include <iostream>
using namespace std;
class student 
{
    public : 
        int rollno; 
        string name; 
    student(int  r ,string n) 
    {
        rollno=r;
        name=n;
        cout<<"parameterized constructor called"<<endl;
    }
    void show()
    {
        cout<<"rollno is  : "<<rollno<<endl;
        cout<<"name is : "<<name<<endl;
    }
};
int main()
{
    student s1(1,"ram");
    s1.show(); 
    return 0;
}
